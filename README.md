# React Awesome Testimonial

## Usage

Installation:
```
yarn add react-awesome-testimonial 
```

or 

```
npm i react-awesome-testimonial
```

### Basic usage

```jsx
import AwesomeTestimonial from 'react-awesome-testimonial';

const app = (
  <AwesomeSlider
    testimonials={[
        {
            name: "Abigail",
            company: "Netflix",
            img_src: "image-url",
            review:
            "Review Text",
        },
        {
            name: "Abigail 2",
            company: "Netflix",
            img_src: "image-url",
            review:
            "Review Text",
        },
    ]}
  />
);
```
